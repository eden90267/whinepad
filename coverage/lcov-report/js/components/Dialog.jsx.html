import invariant from "invariant";
import classNames from 'classnames';
import Button from "./Button";
 
type Props = {
  header: string,
  confirmLabel: string,
  modal: boolean,
  onAction: Function,
  hasCancel: ?boolean,
  children?: Array<any>,
}
 
class Dialog extends Component {
 
  props: Props;

    confirmLabel: 'ok',

    },

  };
